---
title: 'Get Patient'
api: 'GET /services/core/v1/patients/{id}'
description: 'Get a patient by ID'
---

## Path Parameters

<ParamField path="id" type="string" required>
  The ID of the patient to get
</ParamField>

## Response

<ResponseField name="success" type="boolean" required>
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="object" required>
  The patient object
  <Expandable title="Patient properties">
    <ResponseField name="id" type="string" required>
      The ID of the patient
    </ResponseField>
    <ResponseField name="name" type="string" required>
      The name of the patient
    </ResponseField>
    <ResponseField name="client_id" type="string" required>
      The ID of the client that owns this patient
    </ResponseField>
    <ResponseField name="species" type="string">
      The species of the patient
    </ResponseField>
    <ResponseField name="breed" type="string">
      The breed of the patient
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="message" type="string">
  Additional information about the response
</ResponseField>

<RequestExample>
```bash Example Request
curl -X GET 'https://partner-api.getoliver.com/services/core/v1/patients/pat_123' \
-H 'Authorization: Bearer YOUR_TOKEN'
```
</RequestExample>
